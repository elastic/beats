package addfields

import (
	"encoding/json"
	"fmt"

	"github.com/elastic/beats/v7/libbeat/beat"
	conf "github.com/elastic/elastic-agent-libs/config"
	"github.com/elastic/elastic-agent-libs/logp"
	"github.com/elastic/elastic-agent-libs/mapstr"
)

type addFieldsMultipleConfig struct {
	Fields mapstr.M `config:"fields" validate:"required"`
	Target *string  `config:"target"`
}

type addFieldsMultiple struct {
	fields mapstr.M
}

// CreateAddFieldsMultiple constructs an add_fields_multiple processor from config.
// It accepts a list of {fields, target} entries and merges them all into a
// single mapstr.M at construction time, so Run() applies one DeepUpdate.
func CreateAddFieldsMultiple(c *conf.C, _ *logp.Logger) (beat.Processor, error) {
	var configs []addFieldsMultipleConfig
	err := c.Unpack(&configs)
	if err != nil {
		return nil, fmt.Errorf("fail to unpack the add_fields_multiple configuration: %w", err)
	}

	if len(configs) == 0 {
		return nil, fmt.Errorf("add_fields_multiple requires at least one entry")
	}

	merged := mapstr.M{}
	for _, cfg := range configs {
		target := optTarget(cfg.Target, FieldsKey)
		if target != "" {
			merged.DeepUpdate(mapstr.M{target: cfg.Fields})
		} else {
			merged.DeepUpdate(cfg.Fields)
		}
	}

	return &addFieldsMultiple{fields: merged}, nil
}

func (afm *addFieldsMultiple) Run(event *beat.Event) (*beat.Event, error) {
	if event == nil || len(afm.fields) == 0 {
		return event, nil
	}

	event.DeepUpdate(afm.fields.Clone())
	return event, nil
}

func (afm *addFieldsMultiple) String() string {
	s, _ := json.Marshal(afm.fields)
	return fmt.Sprintf("add_fields_multiple=%s", s)
}
